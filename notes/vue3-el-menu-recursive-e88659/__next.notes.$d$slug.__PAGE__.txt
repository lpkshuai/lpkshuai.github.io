### 1. 外层菜单组件：

LeftMenu.vue

```js
<template>
  <el-menu
    :default-active="activeMenu"
    router
    :class="'menu-left'"
    :default-openeds="openedsArr"
    text-color="#fff"
  >
    <LeftSubMenu :menuData="menuData"></LeftSubMenu>
  </el-menu>
</template>

<script setup>
import LeftSubMenu from "@/components/LeftSubMenu.vue";
import { computed } from "vue";
import { useRouter } from "vue-router";

const openedsArr = props.menuData.map((item) => {
  return item.path;
});

const props = defineProps({
  menuData: {
    type: Array,
    default: [],
  },
});

const activeMenu = computed(() => {
  const router = useRouter();
  const { meta, path } = router.currentRoute.value;
  if (meta.matchPath2) {
    return meta.matchPath2;
  } else {
    return path;
  }
});
</script>
```

### 2. 子菜单组件（递归主体）：

LeftSubMenu.vue

```js
<template>
  <template v-for="item in menuData">
    <el-sub-menu
      :key="item.path"
      v-if="item.children && item.children.length > 0"
      :index="item.path"
      :class="item.icon ? '' : 'noIcon'"
    >
      <template #title>
        <img
          class="icon pd-r-10"
          :src="curRoute.indexOf(item.path) != -1 ? item.iconActive : item.icon"
        />
        <img class="icon pd-r-10" :src="item.icon" />
        <span>{{ item.label }}</span>
      </template>
      <LeftSubMenu :menuData="item.children"></LeftSubMenu>
    </el-sub-menu>
    <el-menu-item
      :key="item.id"
      v-else
      :index="item.path"
      :disabled="item.disabled"
    >
      <template #title>
        <img class="icon pd-r-10" :src="item.icon" />
        <span>{{ item.label }}</span>
      </template>
    </el-menu-item>
  </template>
</template>

<script setup>
import LeftSubMenu from "./LeftSubMenu.vue";
import { computed } from "vue";
import { useRouter } from "vue-router";

const props = defineProps({
  menuData: {
    type: Array,
    default: [],
  },
});

const curRoute = computed(() => {
  const router = useRouter();
  const { path } = router.currentRoute.value;
  return path;
});
</script>
```
