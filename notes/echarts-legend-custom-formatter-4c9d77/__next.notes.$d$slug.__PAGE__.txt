  orient: "vertical",
  title: {
    text: "一般公共服务支出构成",

      let total = 0;
      let target;
      for (let i = 0, l = data.length; i < l; i++) {
        total += data[i].value;
        if (data[i].name == name) {
          target = data[i].value;
        }
      }
      const percent = ((target / total) * 100).toFixed(2);
      return `${name}${target}亿元，占比${percent}%`;
